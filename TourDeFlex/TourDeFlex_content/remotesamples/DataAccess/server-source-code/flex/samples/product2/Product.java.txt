package flex.samples.product2;
import java.io.Serializable;
/**
 * Product.java
 * 
 * @author Holly Schinsky - Rich Desktop Solutions, Inc
 * 
 * This class is mapped to from the client side to be used in passing the custom object data
 * in a serialized fashion between client and server (via AMF). The client side must match the same fields
 * and names exactly for the direct serialization to occur.
 * 
 * Created: May 18, 2009
 *
 */
public class Product implements Serializable {

    static final long serialVersionUID = 103844514947365244L;
    
    private int productId;
    private String name;
    private String description;
    private String image;
    private String category;
    private double price;
    private int qtyInStock;
    private String uniqueid;
    
    public Product() {
    	
    }
    
    public Product(int productId, String name, String image, String category, 
    		double price, String description, int qtyInStock, String uniqueid) {
    	
		this.productId = productId;
		this.name = name;
		this.description = description;
		this.image = image;
		this.category = category;
		this.price = price;
		this.qtyInStock = qtyInStock;
		this.uniqueid = uniqueid;
		
	}

    public String getCategory() {
		return category;
	}
	public void setCategory(String category) {
		this.category = category;
	}
	public String getDescription() {
		return description;
	}
	public void setDescription(String description) {
		this.description = description;
	}
	public String getImage() {
		return image;
	}
	public void setImage(String image) {
		this.image = image;
	}
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	public double getPrice() {
		return price;
	}
	public void setPrice(double price) {
		this.price = price;
	}
	public int getProductId() {
		return productId;
	}
	public void setProductId(int productId) {
		this.productId = productId;
	}
	public int getQtyInStock() {
		return qtyInStock;
	}
	public void setQtyInStock(int qtyInStock) {
		this.qtyInStock = qtyInStock;
	}
	public String getUniqueid() {
		return uniqueid;
	}
	public void setUniqueid(String uniqueid) {
		this.uniqueid = uniqueid;
	}

}
